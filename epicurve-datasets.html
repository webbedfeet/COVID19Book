library(sars2pack)
library(DT)
library(dplyr)

8.1 Worldwide datasets

jhu = jhu_data()
DT::datatable(jhu[sample(1:nrow(jhu),500),] %>%
              dplyr::arrange(CountryRegion, date),
              extensions = 'Scroller', options = list(scrollX=TRUE))
